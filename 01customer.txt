create table customer (
	cust_no		number(4)	not null,
	cust_name	varchar(15)	not null,
	email		varchar(35),
	phone		varchar(11),
	birth_date	date,
	gender		char(1),
primary key(cust_no),
constraint chk_email check (REGEXP_LIKE(email, '^[a-zA-Z]\x+@(\S+)$')),
constraint chk_email check (UPPER(gender) in ('M', 'F'))
);
